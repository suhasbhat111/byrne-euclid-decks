  {\Large\textbf{Postulates}}

\noindent\rule{\linewidth}{0.4pt}
\vspace{0.6cm}

\begin{center}
\textbf{I.}\\[6pt]
Let it be granted that a straight line may be drawn from any one point
to any other point.
\end{center}

\vspace{0.8cm}

\begin{center}
\textbf{II.}\\[6pt]
Let it be granted that a finite straight line may be produced to any
length in a straight line.
\end{center}

\vspace{0.8cm}

\begin{center}
\textbf{III.}\\[6pt]
Let it be granted that a circle may be described with any centre at any
distance from that centre.
\end{center}
