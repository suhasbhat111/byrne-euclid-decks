% STATEMENT

\noindent
\begin{minipage}[t]{0.50\linewidth}
\itshape To bisect a given finite straight line
(\drawUnitLine{DB,CD}).
\end{minipage}\hfill
\begin{minipage}[t]{0.46\linewidth}
\centering\drawCurrentPicture
\end{minipage}

\vspace{0.5cm}

% PROOF
\begin{center}
Construct \drawLine[bottom]{AB,CA,CD,DB}~(pr.~I.1),\\
draw \drawUnitLine{AD}, making
$\drawAngle{BAD} = \drawAngle{CAD}$~(pr.~I.9).

\medskip
Then $\drawUnitLine{BD} = \drawUnitLine{DC}$~by~(pr.~I.4),

for~$\drawUnitLine{AB} = \drawUnitLine{AC}$~(const.)
$\drawAngle{BAD} = \drawAngle{CAD}$\\
and \drawUnitLine{AD} common to the two triangles.

Therefore the given line is bisected.
\end{center}

\begin{flushright}Q.\ E.\ D.\end{flushright}
