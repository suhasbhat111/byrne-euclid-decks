\textbf{Book~I.\enspace Prop.\ XV. Theor.}
\medskip\noindent\itshape
If two right lines (\drawUnitLine{AB} and \drawUnitLine{CD}) intersect one
another, the vertical angles \drawAngle{BEC} and \drawAngle{AED},
\drawAngle{CEA} and \drawAngle{DEB} are equal.
\bigskip\noindent\upshape\begin{center}
$\drawAngle{BEC}+\drawAngle{CEA}=\drawTwoRightAngles$\\[4pt]
$\drawAngle{AED}+\drawAngle{CEA}=\drawTwoRightAngles$\\[4pt]
$\therefore \drawAngle{BEC}=\drawAngle{AED}$.\\[6pt]
In the same manner it may be shown that\\
$\drawAngle{CEA}=\drawAngle{DEB}$.
\end{center}
\begin{flushright}Q.\ E.\ D.\end{flushright}
