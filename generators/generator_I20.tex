\textbf{Book~I.\enspace Prop.\ XX. Theor.}
\medskip\noindent\itshape
Any two sides \drawUnitLine{DA} and \drawUnitLine{BD} of a triangle
\drawLine[bottom]{DA,BD,AB} taken together are greater than the third
side (\drawUnitLine{AB}).
\bigskip\noindent\upshape\begin{center}
Produce \drawUnitLine{DA}, and\\
make $\drawUnitLine{CD}=\drawUnitLine{BD}$~(pr.~I.3);\\
draw \drawUnitLine{BC}.\\[6pt]
Then $\because \drawUnitLine{CD}=\drawUnitLine{BD}$~(const.),\\
$\drawAngle{CBD}=\drawAngle{C}$~(pr.~I.5)\\[4pt]
$\therefore \drawAngle{CBD,DBA}>\drawAngle{C}$~(ax.~IX)\\[4pt]
$\therefore \drawUnitLine{DA}+\drawUnitLine{CD}>\drawUnitLine{AB}$~(pr.~I.19)\\[4pt]
and $\therefore \drawUnitLine{DA}+\drawUnitLine{BD}>\drawUnitLine{AB}$.
\end{center}
\begin{flushright}Q.\ E.\ D.\end{flushright}
