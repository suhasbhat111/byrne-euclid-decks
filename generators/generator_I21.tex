\textbf{Book~I.\enspace Prop.\ XXI. Theor.}
\medskip\noindent\itshape
If from any point (\drawPointL[middle]{D}) within a triangle
\drawLine[bottom]{CA,EC,BE,AB}, straight lines be drawn to the
extremities of one side (\drawUnitLine{AB}), these lines must be
together less than the other two sides, but must contain a greater
angle.
\bigskip\noindent\upshape\begin{center}
Produce \drawUnitLine{AD},\\
$\drawUnitLine{CA}+\drawUnitLine{EC}>\drawUnitLine{AD,DE}$~(pr.~I.20),\\
add \drawUnitLine{BE} to each,\\
$\drawUnitLine{CA}+\drawUnitLine{EC,BE}>\drawUnitLine{AD,DE}+\drawUnitLine{BE}$~(ax.~IV)\\[4pt]
in the same manner it may be shown that\\
$\drawUnitLine{AD,DE}+\drawUnitLine{BE}>\drawUnitLine{AD}+\drawUnitLine{BD}$,\\
$\therefore \drawUnitLine{CA}+\drawUnitLine{EC,BE}>\drawUnitLine{AD}+\drawUnitLine{BD}$,\\
which was to be proved.\\[4pt]
Again $\drawAngle{E}>\drawAngle{C}$~(pr.~I.16),\\
and also $\drawAngle{D}>\drawAngle{E}$~(pr.~I.16),\\[4pt]
$\therefore \drawAngle{D}>\drawAngle{C}$.
\end{center}
\begin{flushright}Q.\ E.\ D.\end{flushright}
