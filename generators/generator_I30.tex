\textbf{Book~I.\enspace Prop.\ XXX. Theor.}
\medskip\noindent\itshape
Straight lines (\drawUnitLine{AB} and \drawUnitLine{EF}) which
are parallel to the same straight line (\drawUnitLine{CD}), are
parallel to one another.
\bigskip\noindent\upshape\begin{center}
Let \drawUnitLine{JK} intersect
$\left\{\vcenter{\nointerlineskip\hbox{\drawUnitLine{AB}}%
\nointerlineskip\hbox{\drawUnitLine{CD}}%
\nointerlineskip\hbox{\drawUnitLine{EF}}}\right\}$;\\[6pt]
Then, $\drawAngle{G}=\drawAngle{I}=\drawAngle{H}$~(pr.~I.29),\\[4pt]
$\therefore \drawAngle{G}=\drawAngle{H}$\\[4pt]
$\therefore \drawUnitLine{AB}\parallel\drawUnitLine{EF}$~(pr.~I.28).
\end{center}
\begin{flushright}Q.\ E.\ D.\end{flushright}
