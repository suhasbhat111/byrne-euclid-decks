\textbf{Book~I.\enspace Prop.\ XXXIII. Theor.}
\medskip\noindent\itshape
Straight lines (\drawUnitLine{AC} and \drawUnitLine{BD}) which
join the adjacent extremities of two equal and parallel
straight lines (\drawUnitLine{AB} and \drawUnitLine{CD}), are
themselves equal and parallel.
\bigskip\noindent\upshape\begin{center}
Draw \drawUnitLine{AD} the diagonal.\\[6pt]
$\drawUnitLine{AB}=\drawUnitLine{CD}$~(hyp.)\\
$\drawAngle{BAD}=\drawAngle{CDA}$~(pr.~I.29)\\
and \drawUnitLine{AD} common to the two triangles;\\[4pt]
$\therefore \drawUnitLine{AC}=\drawUnitLine{BD}$,
and $\drawAngle{ADB}=\drawAngle{DAC}$~(pr.~I.4);\\[4pt]
and $\therefore \drawUnitLine{AC}\parallel\drawUnitLine{BD}$~(pr.~I.27).
\end{center}
\begin{flushright}Q.\ E.\ D.\end{flushright}
