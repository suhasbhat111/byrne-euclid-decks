\textbf{Book~I.\enspace Prop.\ XXXIX. Theor.}
\medskip\noindent\itshape
Equal triangles \drawPolygon[bottom]{AEC,ECD} and
\drawPolygon[bottom]{BED,ECD} on the same base
(\drawUnitLine{CD}) and on the same side of it, are between
the same parallels.
\bigskip\noindent\upshape
If \drawUnitLine{AB}, which joins the vertices of the
triangles, be not $\parallel\drawUnitLine{CD}$, draw
$\drawUnitLine{AF}\parallel\drawUnitLine{CD}$~(pr.~I.31),
meeting \drawUnitLine{CG}.\\[4pt]
Draw \drawUnitLine{DF}.\\[4pt]
$\because \drawUnitLine{AF}\parallel\drawUnitLine{CD}$~(const.)\\
$\drawPolygon{AEC,ECD}=\drawPolygon{BED,ECD,FBD}$~(pr.~I.37);\\
but $\drawPolygon{AEC,ECD}=\drawPolygon{BED,ECD}$~(hyp.);\\[4pt]
$\therefore \drawPolygon{BED,ECD}=\drawPolygon{BED,ECD,FBD}$,\\
a part equal to the whole, which is absurd.\\[4pt]
$\therefore \drawUnitLine{AF}\not\parallel\drawUnitLine{CD}$;
and in the same manner it can be demonstrated that no other
line except \drawUnitLine{AB} is $\parallel\drawUnitLine{CD}$;
$\therefore \drawUnitLine{AB}\parallel\drawUnitLine{CD}$.
\begin{flushright}Q.\ E.\ D.\end{flushright}
