\textbf{Book~I.\enspace Prop.\ XLI. Theor.}
\medskip\noindent\itshape
If a parallelogram \drawPolygon[bottom]{ABFG,GFE,AGD,DEG} and a
triangle \drawPolygon[bottom]{GFE,DEG,CFE} are upon the same
base \drawUnitLine{DE} and between the same parallels
\drawUnitLine{AC} and \drawUnitLine{DE}, the parallelogram is
double the triangle.
\bigskip\noindent\upshape\begin{center}
Draw \drawUnitLine{AE} the diagonal.\\[4pt]
Then $\drawPolygon{AGD,DEG}=\drawPolygon{GFE,DEG,CFE}$~(pr.~I.37)\\
$\drawPolygon{ABFG,GFE,AGD,DEG}=\mbox{ twice }\drawPolygon{AGD,DEG}$~(pr.~I.34)\\[4pt]
$\therefore \drawPolygon{ABFG,GFE,AGD,DEG}=\mbox{ twice }\drawPolygon{GFE,DEG,CFE}$.
\end{center}
\begin{flushright}Q.\ E.\ D.\end{flushright}
