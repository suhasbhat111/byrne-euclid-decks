\textbf{Book~I.\enspace Prop.\ XLIII. Theor.}
\medskip\noindent\itshape
The complements \drawPolygon[bottom]{HBGE} and
\drawPolygon[bottom]{FCIE} of the parallelograms which are
about the diagonal of a parallelogram are equal.
\bigskip\noindent\upshape\begin{center}
$\drawPolygon{AEF,FCIE,IDE}=\drawPolygon{AEH,HBGE,GDE}$~(pr.~I.34)\\[4pt]
and $\drawPolygon{AEF,IDE}=\drawPolygon{AEH,GDE}$~(pr.~I.34)\\[4pt]
$\therefore \drawPolygon{FCIE}=\drawPolygon{HBGE}$~(ax.~III).
\end{center}
\begin{flushright}Q.\ E.\ D.\end{flushright}
